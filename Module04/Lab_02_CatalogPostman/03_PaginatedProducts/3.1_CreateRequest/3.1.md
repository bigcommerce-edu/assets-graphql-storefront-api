Request name:
GQL Get Paginated Products

Disable cookie jar:
On

Authorization:
Inherit auth from parent

Request method:
POST

URL:
```
https://store-{{store_hash}}-{{storefront_channel_id}}.mybigcommerce.com/graphql
```

Headers:

| Header       | Value            |
|--------------|------------------|
| Accept       | application/json |
| Content-Type | application/json |

Body:
```
query GetProducts(
    $lastProductCursor: String
) {
    site {
        products(
            after: $lastProductCursor,
            first: 5
        ) {
            pageInfo {
                hasNextPage
                endCursor
            }
            collectionInfo {
                totalItems
            }
            edges {
                cursor
                node {
                    entityId
                    sku
                    name
                    type
                    availabilityV2 {
                        status
                    }
                    inventory {
                        isInStock
                    }
                    variants(isPurchasable: true) {
                        edges {
                            node {
                                entityId
                                inventory {
                                    isInStock
                                }
                            }
                        }
                    }
                }
            }
        }
    }
}
```

Variables:
```
{
    "lastProductCursor": "{{last_product_cursor}}"
}
```
