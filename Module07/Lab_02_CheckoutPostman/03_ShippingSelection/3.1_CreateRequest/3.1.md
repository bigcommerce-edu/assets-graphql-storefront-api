Request name:
GQL Select Shipping

Disable cookie jar:
On

Authorization:
Inherit auth from parent

Request method:
POST

URL:
```
https://store-{{store_hash}}-{{storefront_channel_id}}.mybigcommerce.com/graphql
```

Headers:

| Header       | Value            |
|--------------|------------------|
| Accept       | application/json |
| Content-Type | application/json |

Body:
```
mutation SelectShipping(
    $cartId: String!, 
    $consignmentId: String!, 
    $shippingOptionId: String!
) {
    checkout {
        selectCheckoutShippingOption(
            input: {
                checkoutEntityId: $cartId, 
                consignmentEntityId: $consignmentId, 
                data: {
                    shippingOptionEntityId: $shippingOptionId
                }
            }
        ) {
            checkout {
                shippingConsignments {
                    selectedShippingOption {
                        entityId
                    }
                }
            }
        }
    }
}
```

Variables:
```
{
    "cartId": "{{cart_id}}",
    "consignmentId": "{{consignment_id}}",
    "shippingOptionId": "{{shipping_option_id}}"
}
```
